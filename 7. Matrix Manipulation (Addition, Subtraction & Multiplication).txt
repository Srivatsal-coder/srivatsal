#include <stdio.h>
int main() {
int a[10][10], b[10][10], c[10][10], i, j, k, n;
printf("Enter size of square matrix: ");
scanf("%d", &n);

printf("Enter elements of Matrix A:\n");
for(i = 0; i < n; i++)
for(j = 0; j < n; j++)
scanf("%d", &a[i][j]);
printf("Enter elements of Matrix B:\n");
for(i = 0; i < n; i++)
for(j = 0; j < n; j++)
scanf("%d", &b[i][j]);
// Addition
printf("\nAddition of Matrices:\n");
for(i = 0; i < n; i++) {
for(j = 0; j < n; j++) {
c[i][j] = a[i][j] + b[i][j];
printf("%d ", c[i][j]);
}
printf("\n");
}
// Subtraction
printf("\nSubtraction of Matrices:\n");
for(i = 0; i < n; i++) {
for(j = 0; j < n; j++) {
c[i][j] = a[i][j] - b[i][j];
printf("%d ", c[i][j]);
}
printf("\n");
}
// Multiplication
printf("\nMultiplication of Matrices:\n");
for(i = 0; i < n; i++) {
for(j = 0; j < n; j++) {
c[i][j] = 0;
for(k = 0; k < n; k++)
c[i][j] += a[i][k] * b[k][j];
printf("%d ", c[i][j]);
}
printf("\n");
}
return 0;
}

Output:

Enter elements of Matrix A: 1 2 3 4
Enter elements of Matrix B: 1 0 0 1

Subtraction of Matrices:

Multiplication of Matrices:
