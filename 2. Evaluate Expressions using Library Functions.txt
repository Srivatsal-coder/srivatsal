#include <stdio.h>
#include <math.h>
int main() {
float r=3.0, A=2.0, B=4.0, C=6.0, S=5.0, x=2.0, y=3.0, z=4.0;
printf("πr^2 = %.2f\n", M_PI*r*r);
printf("(A+B+(2C/3A)+A^2+2B) = %.2f\n", A+B+(2*C/(3*A))+pow(A,2)+2*B);
printf("√S(S-A)(S-B)(S-C) = %.2f\n", sqrt(S*(S-A)*(S-B)*(S-C)));
printf("log(x^3+y^3+z^3) = %.2f\n", log(pow(x,3)+pow(y,3)+pow(z,3)));
return 0;
}

Output:

log(x^3+y^3+z^3) = 4.60
